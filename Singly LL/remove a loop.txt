class Solution
{
    //Function to remove a loop in the linked list.
    public static void removeLoop(Node head)
    {
        Node slow=head;
        Node fast=head;
        while(slow!=null && fast!=null && fast.next!=null)
        {
            slow=slow.next;
            fast=fast.next.next;
            
            // if loop is there
            if(slow==fast)
            break;
        }
        // when slow is pointing to starting node of loop itself
        if(slow==head)
        {
            while(fast.next!=slow)
            {
                fast=fast.next;
            }
            fast.next=null;
        }
        // finding starting point of loop which will be slow and then changing next of fast pointer to null
        if(slow==fast)
        {
            slow=head;
            //finding starting point of loop 
            while(slow.next!=fast.next)
            {
                slow=slow.next;
                fast=fast.next;
            }   
            // changing to null
            fast.next=null;
        }

    }
}
