import java.util.*;
class Node
{
  int data;
  Node next;

    Node(int data)
    {
        this.data=data;
        this.next=null;
    }
}

class LL
{
    public void print(Node head)
    {
        Node curr=head;
        while(curr!=null)
        {
            System.out.print(curr.data+"->");
            curr=curr.next;
        }
        System.out.print("null");
        System.out.println();
    }
}
public class Main
{

  public static void main (String[]args)
  {
    Scanner s=new Scanner(System.in);
    LL ll=new LL();
    int n=s.nextInt();
    Node head=new Node(s.nextInt());
    Node tail=head;
    while(n!=1)
      {
	    tail.next= new Node(s.nextInt());
	    tail=tail.next;
	    n--;
      }
    ll.print(head);
  }
}
