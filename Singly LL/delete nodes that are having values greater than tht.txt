class Solution
{
    Node reverse(Node p)
    {
        if(p.next==null)
        {
            Node newhead=p;
            return newhead;
        }
        Node newhead=reverse(p.next);
        Node q=p.next;
        q.next=p;
        p.next=null;
        return newhead;
    }
    Node delete(Node head)
    {
        if(head==null || head.next==null)
        {
            return head;
        }
        
        Node prev=head;
        Node curr=head.next;
        while(curr!=null)
        {
            if(prev.data>curr.data)
            {
                prev.next=curr.next;
                curr=curr.next;
            }
            else
            {
                prev=prev.next;
                curr=curr.next;
            }
        }
        return head;
    }
    Node compute(Node head)
    {
        head=reverse(head);
        head=delete(head);
        head=reverse(head);
        return head;
    }
}
